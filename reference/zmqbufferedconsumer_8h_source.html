50 /**
51  * A buffered consumer that sends data via ZeroMQ.
52  *
53  * The intended use-case is for this class to be used together with
54  * ZeroMQBufferedConsumer to implement distributed producer-consumer setups.
55  * Suppose you have an existing setup that looks like this, with sections of
56  * your code producing items of type T and other sections consuming them:
57  *
58  * [Producer] -> [Consumer]
59  *
60  * Then, assuming that items can be serialized with cereal, the
61  * ZeroMQBufferedConsumer/Producer classes enable the following design:
62  *
63  * [Producer] -> [ZeroMQBufferedConsumer]
64  * |
65  * TCP
66  * |
67  * [ZeroMQBufferedProducer] -> [Consumer]
68  *
69  * As in common::BufferedConsumer, you specify the number of threads and a queue
70  * size. In addition, you supply a list of end points that
71  * ZeroMQBufferedProducer instances have been bound to. Data will be send to
72  * end-points in a round-robin fashion. If producer end points don't accept new
73  * data (because their queue is full and items are not consumed fast enough),
74  * `enqueue()` will eventually block and perform retries.
75  */
76 template <typename T>

99 template <typename T>
100 ZeroMQBufferedConsumer<T>::ZeroMQBufferedConsumer(
101  uint8_t nthreads,
102  size_t maxQueueSize,
103  std::vector<std::string> endpoints,
104  std::shared_ptr<zmq::context_t> context) {
105  client_ = std::make_unique<detail::RRClientWrapper>(
106  maxQueueSize, std::move(endpoints), std::move(context));
107  // BufferedConsumer for sending out data
108  bcsend_ = std::make_unique<common::BufferedConsumer<ClientAction>>(
109  1, 1, [this, maxQueueSize](ClientAction ca) {
110  client_->perform(ca.first, std::move(ca.second));
111  // XXX We can't queue up retries indefinitely, so let's do some busy
112  // waiting with exponential backoff for retries.
113  if (client_->numScheduledForRetry() > maxQueueSize) {
114  auto start = std::chrono::steady_clock::now();
115  int ntry = 0;
116  while (client_->numScheduledForRetry() > maxQueueSize) {
117  if (ntry++ > 0) {
118  std::this_thread::sleep_for(std::chrono::milliseconds(
119  int(10 * std::pow(2, std::min(ntry, 5)))));
120  }
121  client_->perform(
122  RRClientWrapper::Action::SendRetries, std::vector<char>());
123  client_->perform(
124  RRClientWrapper::Action::WaitForReplies, std::vector<char>());
125  }
126  auto elapsed = std::chrono::steady_clock::now() - start;
127  auto ms =
128  std::chrono::duration_cast<std::chrono::milliseconds>(elapsed)
129  .count();
130  VLOG_ALL(1) << "ZeroMQBufferedConsumer: waited " << ms
131  << "ms for retries";
132  }
133  });
134  // BufferedConsumer for data serialization
135  bcser_ = std::make_unique<common::BufferedConsumer<T>>(
136  nthreads, maxQueueSize, [this](T data) {
137  common::OMembuf buf;
138  {
139  common::zstd::ostream os(&buf);
140  cereal::BinaryOutputArchive ar(os);
141  ar(data);
142  }
143  bcsend_->enqueue(
144  std::make_pair(RRClientWrapper::Action::Send, buf.takeData()));
145  });
146 }
