36 /**
37  * A Behavior is a self-contained situational micro rule.
38  */
39 class Behavior {
40  public:
41  /// Checks if a unit still needs a micro decision. If so, invokes onPerform()
42  void perform(Agent& agent);
43 
44  virtual ~Behavior() = default;
45 
46  protected:
47  /// Convenience method: Form a MicroAction reflecting a decision to
48  /// issue a UPC for this unit.
49  MicroAction doAction(std::shared_ptr<UPCTuple> upc) const {
50  return MicroAction{true, upc};
51  }
52  /// Convenience: A MicroAction reflecting a decision to
53  /// ignore this unit and let another Behavior control it.
54  const MicroAction pass = MicroAction();
55 
56  /// Convenience: A MicroAction reflecting a decision to
57  /// do nothing with this unit and let no other Behavior control it.
58  const MicroAction doNothing = doAction(nullptr);
59 
60  /// Decide what to do with a unit that has not yet been controlled by a
61  /// Behavior.
62  virtual MicroAction onPerform(Agent&) = 0;
63 };
