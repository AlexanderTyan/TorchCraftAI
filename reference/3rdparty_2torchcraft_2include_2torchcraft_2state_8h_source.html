153  int getUpgradeLevel(BW::UpgradeType ut) {
154  if (!(frame->resources[player_id].upgrades & (1ll << ut))) {
155  return 0;
156  }
157  auto constexpr NB_LVLABLE_UPGRADES = 16;
158  if (ut >= NB_LVLABLE_UPGRADES) {
159  return 1;
160  }
161  uint64_t lvls = frame->resources[player_id].upgrades_level;
162  if (lvls & (1ll << ut)) {
163  return 2;
164  }
165  if (lvls & (1ll << (ut + NB_LVLABLE_UPGRADES))) {
166  return 3;
167  }
168  return 1;
169  }
