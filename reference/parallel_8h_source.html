21 /**
22  * A simple producer/consumer class.
23  *
24  * This class is dead-simple, but sometimes useful. You specify the element type
25  * for the queue in the type, and then instantiate it with functor which will
26  * run in a separate thread. The main function of the class is enqueue(), which,
27  * well, adds stuff to the queue. You also specify a maximum queue size on
28  * construction; if that size is reached, enqueue() will block.
29  *
30  * As a special case, you can use this class with 0 threads. This means that the
31  * supplied functor will be called directly in the thread calling enqueue().
32  * Items will be buffered implicitly by enqueue() blocking until consumption.
33  *
34  * If you want to wait for the consumers to finish, call wait(). If you want to
35  * stop the consumer threads, destruct the object.
36  *
37  * The implementation assumes that objects of type T are in a valid state (i.e.
38  * can be destructed) after moving. If that's not the case for your type, go fix
39  * your type.
40  */
41 template <typename T>
42 class BufferedConsumer {
43  using Function = std::function<void(T)>;
44 
45  public:
46  using type = T;
47 
48  BufferedConsumer(uint8_t nthreads, size_t maxQueueSize, Function&& fn);
49 
50  /// Stops the consumers, discarding any items in the queue
51  ~BufferedConsumer();
52 
53  /// Blocks until the queue is empty or the consumers are stopped
54  void wait();
55 
56  /// Adds another item to the work queue, possibly blocking
57  /// If the number of threads is zero, execute directly in the calling thread's
58  /// context (and thus block).
59  void enqueue(T arg);
60 
61  void run();
62 
63  private:
64  size_t const maxQueueSize_;
65  bool stop_ = false;
66  int64_t consuming_ = 0;
67  Function fn_;
68  std::vector<std::thread> threads_;
69  std::queue<T> queue_;
70  std::mutex mutex_;
71  std::condition_variable itemReady_;
72  std::condition_variable itemDone_;
73 };

76  * A simple producer class.
77  *
78  * You specify a function that will generate data for you somehow, ending when
79  * it returns an optional without a value, and this producer will multithread it
80  * for you automatically. The function should be thread-safe, and data is not
81  * guaranteed to arrive in the same order it was generated in, unless you do it
82  * yourself. If you want to stop the consumer threads, destruct the object. If
83  * you try destructing the object while get() is still being called, it will
84  * result in a runtime error.
85  */

95

97  ~BufferedProducer();

99  std::optional<T> get();

155 template <typename T>
156 void BufferedConsumer<T>::enqueue(T arg) {
157  if (!threads_.empty()) {
158  {
159  std::unique_lock<std::mutex> lock(mutex_);
160  itemDone_.wait(
161  lock, [&] { return stop_ || queue_.size() < maxQueueSize_; });
162  if (stop_) {
163  throw std::runtime_error("BufferedConsumer not active");
164  }
165  queue_.push(std::move(arg));
166  }
167  itemReady_.notify_one();
168  } else {
169  {
170  std::unique_lock<std::mutex> lock(mutex_);
171  if (stop_) {
172  throw std::runtime_error("BufferedConsumer not active");
173  }
174  consuming_++;
175  fn_(std::move(arg));
176  consuming_--;
177  }
178  itemDone_.notify_all();
179  }
180 }

182 template <typename T>
183 void BufferedConsumer<T>::run() {
184  std::unique_lock<std::mutex> lock(mutex_);
185  while (true) {
186  itemReady_.wait(lock, [&] { return stop_ || !queue_.empty(); });
187  if (stop_) {
188  break;
189  }
190  if (queue_.empty()) {
191  continue;
192  }
193 
194  T item = std::move(queue_.front());
195  queue_.pop();
196 
197  consuming_++;
198  lock.unlock();
199  fn_(std::move(item));
200  lock.lock();
201  consuming_--;
202 
203  // Only remove the item from the queue once it has been consumed
204  // Ideally we'd do the notification without holding the lock, but doing so
205  // we save one lock/unlock cycle. Let's trust implementations to recognize
206  // this scenario (cf.
207  // https://en.cppreference.com/w/cpp/thread/condition_variable/notify_one)
208  itemDone_.notify_all();
209  }
210 }

244 template <typename T>
245 std::optional<T> BufferedProducer<T>::get() {
246  std::unique_lock<std::mutex> lock(mutex_);
247  queueCV_.wait(
248  lock, [&] { return stop_ || !queue_.empty() || running_ == 0; });
249  if (stop_) {
250  throw std::runtime_error("BufferedProducer not active");
251  }
252  if (running_ == 0 && queue_.empty()) {
253  return std::optional<T>();
254  }
255  auto ret = queue_.front().get();
256  queue_.pop();
257  queueCV_.notify_all();
258  return ret;
259 }

261 template <typename T>
262 void BufferedProducer<T>::run(Function fn) {
263  while (true) {
264  std::unique_lock<std::mutex> lock(mutex_);
265  queueCV_.wait(lock, [&] {
266  return stop_ || queue_.size() + working_ < maxQueueSize_;
267  });
268  if (stop_) {
269  break;
270  }
271 
272  std::promise<std::optional<T>> dataPromise;
273  working_++;
274 
275  lock.unlock();
276  auto result = fn();
277  bool done = (!result.has_value());
278  lock.lock();
279 
280  working_--;
281  if (done) {
282  running_--;
283  queueCV_.notify_all();
284  break;
285  }
286  dataPromise.set_value(std::move(result));
287  queue_.push(dataPromise.get_future());
288  queueCV_.notify_all();
289  }
