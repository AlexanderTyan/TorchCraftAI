32 /**
33  * Input:
34  * - getter: for the list of objects to compare to
35  * - valid: f(agent, position, obj) => bool for whether this position is valid
36  * given the obj
37  * - score: f(agent, position, obj) => float to score this position
38  *
39  * The filter will decide if a position is valid by combining the valid func
40  * and score function according to the PositionFilterPolicy.
41  */
42 template <typename T, typename Container>
43 class FuncPositionFilter : public PositionFilter {
44  public:
45  FuncPositionFilter(
46  std::function<Container const && (Unit*)> getter,
47  std::function<bool(Unit*, Position const&, T)> valid,
48  std::function<float(Unit*, Position const&, T)> scoreFunc,
49  PositionFilterPolicy policy = PositionFilterPolicy::ACCEPT_IF_ALL,
50  bool blocking = false)
51  : getter_(std::move(getter)),
52  valid_(std::move(valid)),
53  score_(std::move(scoreFunc)),
54  policy_(policy),
55  blocking_(blocking) {}
56 
57  bool isValid(Unit* agent, Position const& pos) override {
58  VLOG(4) << "PositionFilter: in filter valid";
59  auto objects = getter_(agent);
60  VLOG(4) << "PositionFilter: in valid non empty set? size is: "
61  << objects.size();
62  if (policy_ == PositionFilterPolicy::ACCEPT_IF_ALL) {
63  for (auto obj : objects) {
64  if (!valid_(agent, pos, obj)) {
65  return false;
66  }
67  }
68  return true;
69  } else if (policy_ == PositionFilterPolicy::ACCEPT_IF_ANY) {
70  for (auto obj : objects) {
71  if (valid_(agent, pos, obj)) {
72  return true;
73  }
74  }
75  return false; // false if empty: must be valid for some
76  } else {
77  LOG(DFATAL) << "MoveHelpers: incorrect position filter policy";
78  }
79  return false;
80  }
81 
82  float score(Unit* agent, Position const& pos) override {
83  if (policy_ == PositionFilterPolicy::ACCEPT_IF_ANY) {
84  auto best_score = kfInfty;
85  for (auto obj : getter_(agent)) {
86  auto s = score_(agent, pos, obj);
87  if (s < best_score) {
88  best_score = s;
89  }
90  }
91  return best_score;
92  } else if (policy_ == PositionFilterPolicy::ACCEPT_IF_ALL) {
93  auto best_score = -kfInfty;
94  for (auto obj : getter_(agent)) {
95  auto s = score_(agent, pos, obj);
96  if (s > best_score) {
97  best_score = s;
98  }
99  }
100  return best_score;
101  } else {
102  LOG(DFATAL) << "MoveHelpers: incorrect position filter policy";
103  return 0;
104  }
105  }
106 
107  bool blocking() override {
108  return blocking_;
109  }
110 
111  protected:
112  std::function<Container const && (Unit*)> getter_;
113  std::function<bool(Unit*, Position const&, T)> valid_;
114  std::function<float(Unit*, Position const&, T)> score_;
115  PositionFilterPolicy policy_;
116  bool blocking_;
117 };

160 class UnionPositionFilter : public PositionFilter {
161  public:
162  UnionPositionFilter(
163  PositionFilters l,
164  PositionFilterPolicy policy = PositionFilterPolicy::ACCEPT_IF_ALL,
165  bool blocking = false)
166  : allFilters_(l), policy_(policy), blocking_(blocking) {}
167 
168  bool isValid(Unit* agent, Position const& pos) override {
169  if (policy_ == PositionFilterPolicy::ACCEPT_IF_ALL) {
170  for (auto& filt : allFilters_) {
171  if (!filt->isValid(agent, pos)) {
172  return false;
173  }
174  }
175  return true;
176  } else if (policy_ == PositionFilterPolicy::ACCEPT_IF_ANY) {
177  for (auto& filt : allFilters_) {
178  if (filt->isValid(agent, pos)) {
179  return true;
180  }
181  }
182  return false;
183  }
184  LOG(ERROR) << "invalid moveFilter policy";
185  return false;
186  }
187 
188  float score(Unit* agent, Position const& pos) override {
189  if (policy_ == PositionFilterPolicy::ACCEPT_IF_ALL) {
190  float max_score = -kfInfty;
191  for (auto& filt : allFilters_) {
192  auto score = filt->score(agent, pos);
193  if (score > max_score) {
194  max_score = score;
195  }
196  }
197  return max_score;
198  } else if (policy_ == PositionFilterPolicy::ACCEPT_IF_ANY) {
199  float min_score = kfInfty;
200  for (auto& filt : allFilters_) {
201  auto score = filt->score(agent, pos);
202  if (score < min_score) {
203  min_score = score;
204  }
205  }
206  return min_score;
207  }
208  LOG(ERROR) << "invalid moveFilter policy";
209  return -kfInfty;
210  }
211 
212  bool blocking() override {
213  return blocking_;
214  }
215 
216  protected:
217  PositionFilters allFilters_;
218  PositionFilterPolicy policy_;
219  bool blocking_;
