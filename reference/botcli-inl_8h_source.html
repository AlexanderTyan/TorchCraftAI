77 void setupPlayerFromCli(Player* player) {
78  // Configure according to flags defined above
79  player->setFrameskip(FLAGS_frameskip);
80  player->setRealtimeFactor(FLAGS_realtime_factor);
81  player->setWarnIfSlow(FLAGS_warn_if_slow);
82  player->setNonBlocking(!FLAGS_blocking);
83  player->setCheckConsistency(FLAGS_consistency);
84  player->setCollectTimers(FLAGS_timers);
85  player->setLogFailedCommands(FLAGS_log_failed_commands);
86  player->setDraw(FLAGS_draw);
87  player->setMapHack(FLAGS_map_hack);
88 
89  // Add modules
90  player->addModule(Module::make(kAutoTopModule));
91  for (auto name : utils::stringSplit(FLAGS_modules, ',')) {
92  if (!name.empty()) {
93  player->addModule(Module::make(name));
94  }
95  }
96  player->addModule(Module::make(kAutoBottomModule));
97  if (!FLAGS_trace_along_replay_file.empty()) {
98  player->dumpTraceAlongReplay(FLAGS_trace_along_replay_file);
99  }
100 }
